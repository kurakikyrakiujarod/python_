data1 = [random.gauss(15,10) for i in range(500)]

plt.legend(loc='best')

# 正太分布，又称高斯分布。其特征为中间高两边低左右对称。它有以下几个性质：
# 集中性：曲线的最高峰位于正中央，且位置为均数所在的位置。
# 对称性：正太分布曲线以均数所在的位置为中心左右对称且曲线两段无线超近于横轴。
# 均匀变动性：正太分布曲线以均数所在的位置为中心均匀向左右两侧下降。

R = X**2
R_sum = R.sum(axis = 1)
