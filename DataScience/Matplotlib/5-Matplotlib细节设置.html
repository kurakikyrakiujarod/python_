import math
plt.figure(figsize=(20,12),dpi=80)
x = np.random.normal(loc=0.0,scale=1.0,size=300)
width= 0.5
# print(x)
# print(math.floor(x.min()))
# print(math.ceil(x.max()))
bins = np.arange(math.floor(x.min()-width),math.ceil(x.max())+width,width)
# print(bins)
ax = plt.subplot(111)
ax.spines['top'].set_visible(False)
ax.spines['right'].set_visible(False)
# ax.spines['left'].set_visible(False)
# ax.spines['bottom'].set_visible(False)
# plt.tick_params(bottom=False,top=False,left=False,right=False,colors='red')
plt.grid()
plt.hist(x,alpha=0.5,bins=bins)
plt.show()
