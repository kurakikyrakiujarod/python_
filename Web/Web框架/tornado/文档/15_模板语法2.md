2.py文件中传值
```
#自定义对象
class Person(object):
    def __init__(self,uname):
        self.uname = uname
#倒序方法
def reverse(o):
    if isinstance(o,list):
        o.reverse()
    return o


class IndexHandler(tornado.web.RequestHandler):
    def get(self, *args, **kwargs):
        # 对象
        p = Person('xiaohua')
        # 列表
        l = ['1', '2', '3']
        # 字典
        d = {'d1': 'm1', 'd2': 'm2', 'd3': 'm3'}

        # 转义字符
        str = '<h3>hahaha</h3>'

        self.render('templates/index2.html', name='xiaoming', p=p, l=l, d=d, r=reverse, str=str)
```
